28 template< uint32 _SH >
29 class TBlockBlender_RGBA8_SSE
30 {
31 public:
32  static void ProcessScanLineSSE( TBlock< _SH >* iBlockTop
33  , TBlock< _SH >* iBlockBack
34  , typename TBlock< _SH >::tPixelType iOpacity
35  , const int iLine
36  , const int iX1
37  , const int iX2
38  , const FPoint& iShift
39  , uint8 iAlphaIndex )
40  {
41  // Base ptrs for scanlines
42  uint8* bpb = iBlockBack->PixelPtr( iX1, iLine );
43  uint8* bpt = iBlockTop->PixelPtr( iX1 + iShift.x, iLine + iShift.y );
44  const int delta = iX2 - iX1;
45  const int op = delta / 4;
46 
47  uint8 asmi[4];
48  asmi[0] = iAlphaIndex;
49  asmi[1] = iAlphaIndex + 4;
50  asmi[2] = iAlphaIndex + 8;
51  asmi[3] = iAlphaIndex + 12;
52  __m128i asmv = _mm_set_epi8( asmi[3], asmi[3], asmi[3], asmi[3]
53  , asmi[2], asmi[2], asmi[2], asmi[2]
54  , asmi[1], asmi[1], asmi[1], asmi[1]
55  , asmi[0], asmi[0], asmi[0], asmi[0] );
56  __m128i opa = _mm_set1_epi8( iOpacity );
57  __m128i max = _mm_set1_epi8( 255 );
58 
59  for( int i = 0; i < op; ++i )
60  {
61  __m128i pb = _mm_loadu_si128( (const __m128i*)bpb );
62  __m128i pt = _mm_loadu_si128( (const __m128i*)bpt );
63  __m128i ab = _mm_shuffle_epi8( pb, asmv );
64  __m128i at = _mm_shuffle_epi8( pt, asmv );
65  __m128i zero = _mm_setzero_si128();
66  __m128i atmullo = _mm_mullo_epi16( _mm_unpacklo_epi8( at, zero), _mm_unpacklo_epi8( opa, zero));
67  __m128i atmulhi = _mm_mullo_epi16( _mm_unpackhi_epi8( at, zero), _mm_unpackhi_epi8( opa, zero));
68  __m128i ones = _mm_set1_epi16( 1 );
69  __m128i atmullodown = _mm_srli_epi16( _mm_adds_epu16( _mm_adds_epu16( atmullo, ones ), _mm_srli_epi16( atmullo, 8 ) ), 8 );
70  __m128i atmulhidown = _mm_srli_epi16( _mm_adds_epu16( _mm_adds_epu16( atmulhi, ones ), _mm_srli_epi16( atmulhi, 8 ) ), 8 );
71  __m128i absplo = _mm_unpacklo_epi8( ab, zero);
72  __m128i absphi = _mm_unpackhi_epi8( ab, zero);
73  __m128i addabatsplo = _mm_add_epi16( absplo, atmullodown );
74  __m128i addabatsphi = _mm_add_epi16( absphi, atmulhidown );
75  __m128i mulabatlo = _mm_mullo_epi16( absplo, atmullodown );
76  __m128i mulabathi = _mm_mullo_epi16( absphi, atmulhidown );
77  __m128i mulanatlodown = _mm_srli_epi16( _mm_adds_epu16( _mm_adds_epu16( mulabatlo, ones ), _mm_srli_epi16( mulabatlo, 8 ) ), 8 );
78  __m128i mulanathidown = _mm_srli_epi16( _mm_adds_epu16( _mm_adds_epu16( mulabathi, ones ), _mm_srli_epi16( mulabathi, 8 ) ), 8 );
79  __m128i resabatlosp = _mm_sub_epi16( addabatsplo, mulanatlodown );
80  __m128i resabathisp = _mm_sub_epi16( addabatsphi, mulanathidown );
81  __m128i resmasklo = _mm_set_epi8(0x80, 0x80, 0x80, 0x80, 0x80, 0x80, 0x80, 0x80, 14, 12, 10, 8, 6, 4, 2, 0);
82  __m128i resmaskhi = _mm_set_epi8(14, 12, 10, 8, 6, 4, 2, 0, 0x80, 0x80, 0x80, 0x80, 0x80, 0x80, 0x80, 0x80);
83  __m128i ar = _mm_or_si128( _mm_shuffle_epi8( resabatlosp, resmasklo ), _mm_shuffle_epi8( resabathisp, resmaskhi ) );
84  __m128i ex32psm = _mm_set_epi8( 0x80, 0x80, 0x80, 0x80, 0x80, 0x80, 0x80, 0x80, 0x80, 0x80, 0x80, 0x80, 12, 8, 4, 0 );
85  __m128i arshpr32 = _mm_shuffle_epi8( ar, ex32psm );
86  __m128i atcat = _mm_or_si128( _mm_shuffle_epi8( atmullodown, resmasklo ), _mm_shuffle_epi8( atmulhidown, resmaskhi ) );
87  __m128i atshpr32 = _mm_shuffle_epi8( atcat, ex32psm );
88  __m128i ar32 = _mm_cvtepu8_epi32( arshpr32 );
89  __m128i at32 = _mm_cvtepu8_epi32( atshpr32 );
90  __m128 arps = _mm_cvtepi32_ps( ar32 );
91  __m128 atps = _mm_cvtepi32_ps( at32 );
92  __m128 maxps = _mm_set1_ps( 255.f );
93  __m128 avarpsuns = _mm_mul_ps( _mm_div_ps( atps, arps ), maxps ) ;
94  __m128 zerops = _mm_setzero_ps();
95  __m128 atzerom = _mm_cmpeq_ps( atps, zerops );
96  __m128 avarps = _mm_blendv_ps( atps, avarpsuns, atzerom );
97  __m128i avarsh32 = _mm_cvtps_epi32( avarps );
98  __m128i shv = _mm_set_epi8( 12, 12, 12, 12, 8, 8, 8, 8, 4, 4, 4, 4, 0, 0, 0, 0 );
99  __m128i avar = _mm_shuffle_epi8( avarsh32, shv );
100  __m128i invavar = _mm_sub_epi8( max, avar );
101  __m128i invavarlo = _mm_unpacklo_epi8( invavar, zero);
102  __m128i invavarhi = _mm_unpackhi_epi8( invavar, zero);
103  __m128i pbsplo = _mm_unpacklo_epi8( pb, zero);
104  __m128i pbsphi = _mm_unpackhi_epi8( pb, zero);
105  __m128i mulinvavarpbsplo = _mm_mullo_epi16( invavarlo, pbsplo);
106  __m128i mulinvavarpbsphi = _mm_mullo_epi16( invavarhi, pbsphi );
107  __m128i invab = _mm_sub_epi8( max, ab );
108  __m128i invablo = _mm_unpacklo_epi8( invab, zero );
109  __m128i invabhi = _mm_unpackhi_epi8( invab, zero );
110  __m128i ptsplo = _mm_unpacklo_epi8( pt, zero);
111  __m128i ptsphi = _mm_unpackhi_epi8( pt, zero);
112  __m128i mulinvabptlo = _mm_mullo_epi16( invablo, ptsplo);
113  __m128i mulinvabpthi = _mm_mullo_epi16( invabhi, ptsphi );
114  __m128i mulabbllo = _mm_mullo_epi16( absplo, ptsplo); // BLEND
115  __m128i mulabblhi = _mm_mullo_epi16( absphi, ptsphi); // BLEND
116  __m128i rmlo = _mm_add_epi16( mulinvabptlo, mulabbllo );
117  __m128i rmhi = _mm_add_epi16( mulinvabpthi, mulabblhi );
118  __m128i rmlodown = _mm_srli_epi16( _mm_adds_epu16( _mm_adds_epu16( rmlo, ones ), _mm_srli_epi16( rmlo, 8 ) ), 8 );
119  __m128i rmhidown = _mm_srli_epi16( _mm_adds_epu16( _mm_adds_epu16( rmhi, ones ), _mm_srli_epi16( rmhi, 8 ) ), 8 );
120  __m128i avarlo = _mm_unpacklo_epi8( avar, zero);
121  __m128i avarhi = _mm_unpackhi_epi8( avar, zero);
122  __m128i rrlo = _mm_mullo_epi16( avarlo, rmlodown );
123  __m128i rrhi = _mm_mullo_epi16( avarhi, rmhidown );
124  __m128i cenlo = _mm_add_epi16( mulinvavarpbsplo, rrlo );
125  __m128i cenhi = _mm_add_epi16( mulinvavarpbsphi, rrhi );
126  __m128i cenlodown = _mm_srli_epi16( _mm_adds_epu16( _mm_adds_epu16( cenlo, ones ), _mm_srli_epi16( cenlo, 8 ) ), 8 );
127  __m128i cenhidown = _mm_srli_epi16( _mm_adds_epu16( _mm_adds_epu16( cenhi, ones ), _mm_srli_epi16( cenhi, 8 ) ), 8 );
128  __m128i res = _mm_or_si128( _mm_shuffle_epi8( cenlodown, resmasklo ), _mm_shuffle_epi8( cenhidown, resmaskhi ) );
129  _mm_storeu_si128( (__m128i*)bpb, res );
130  bpb += 16;
131  bpt += 16;
132  }
133  }

135  static void Run( TBlock< _SH >* iBlockTop
136  , TBlock< _SH >* iBlockBack
137  , typename TBlock< _SH >::tPixelType iOpacity
138  , const FRect& iROI
139  , const FPoint& iShift
140  , const FPerformanceOptions& iPerformanceOptions= FPerformanceOptions() )
141  {
142  const int x1 = iROI.x;
143  const int y1 = iROI.y;
144  const int x2 = x1 + iROI.w;
145  const int y2 = y1 + iROI.h;
146  uint8 alpha_index = (uint8) TBlock< _SH >::tPixelProxy::RedirectedIndex( 3 );
147  FThreadPool& global_pool = FGlobalThreadPool::Get();
148  for( int y = y1; y < y2; ++y )
149  global_pool.ScheduleJob( TBlockBlender_RGBA8_SSE< _SH >::ProcessScanLineSSE, iBlockTop, iBlockBack, iOpacity, y, x1, x2, iShift, alpha_index );
150 
151  global_pool.WaitForCompletion();
152  }
