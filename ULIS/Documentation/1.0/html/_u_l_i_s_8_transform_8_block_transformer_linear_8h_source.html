75  {
76  using tPixelType = typename TBlock< _SH >::tPixelType;
77  using tPixelValue = typename TBlock< _SH >::tPixelValue;
78  using tPixelProxy = typename TBlock< _SH >::tPixelProxy;
79  using tPixelBase = typename TPixelBase< _SH >;
80  using info = TBlockInfo< _SH >;
81 
82  const int x1 = 0;
83  const int y1 = 0;
84  const int x2 = iDstBlock->Width();
85  const int y2 = iDstBlock->Height();
86  const int maxx = iSrcBlock->Width();
87  const int maxy = iSrcBlock->Height();
88  tPixelValue fallback = tPixelValue();
89  fallback.SetAlpha( 0 );
90  for( int y = y1; y < y2; ++y )
91  {
92  for( int x = x1; x < x2; ++x )
93  {
94  glm::vec2 point_in_src = glm::vec2( iInverseTransform * glm::vec3( x, y, 1.f ) ); // - glm::vec2( 0.5, 0.5 );
95  int left = floor( point_in_src.x );
96  int up = floor( point_in_src.y );
97  int right = left + 1;
98  int bot = up + 1;
99  float tx = point_in_src.x - float( left );
100  float ty = point_in_src.y - float( up );
101 
102  auto lerp = [&]( const tPixelBase& iA, const tPixelBase& iB, float t ) {
103  tPixelValue ret;
104  float al_a = ConvType< tPixelType, float >( iA.GetAlpha() );
105  float al_b = ConvType< tPixelType, float >( iB.GetAlpha() );
106  float al_c = ( al_a * ( 1.f - t ) + al_b * ( t ) );
107  ret.SetAlpha( ConvType< float, tPixelType >( al_c ) );
108  for( int i = 0; i < info::_nf._nc; ++i ) {
109  tPixelType el_a = iA.GetComponent( i ) * al_a;
110  tPixelType el_b = iB.GetComponent( i ) * al_b;
111  tPixelType el_r = static_cast< tPixelType >( ( el_a * ( 1.f - t ) + el_b * ( t ) ) / al_c );
112  ret.SetComponent( i, el_r );
113  }
114  return ret;
115  };
116  #define TEMP( iX, iY ) ( iX < 0 || iY < 0 || iX >= maxx || iY >= maxy ) ? fallback : iSrcBlock->PixelValue( iX, iY );
117  const tPixelValue c00 = TEMP( left, up );
118  const tPixelValue c10 = TEMP( right, up );
119  const tPixelValue c11 = TEMP( right, bot );
120  const tPixelValue c01 = TEMP( left, bot );
121  const tPixelValue a = lerp( c00, c10, tx );
122  const tPixelValue b = lerp( c01, c11, tx );
123  const tPixelValue c = lerp( a, b, ty );
124  iDstBlock->SetPixelValue( x, y, c );
125  }
126  }
127  }
