215  void Transform( const glm::mat3& mat )
216  {
217  std::vector< glm::vec3 > src_points;
218  std::vector< glm::vec3 > dst_points;
219  GetPoints( &src_points );
220  // We compute result points by multiplicating mat with vec3s
221  // It is important that mat is on the left-hand side
222  // because glm mats are column major.
223  dst_points.push_back( mat * src_points[0] );
224  dst_points.push_back( mat * src_points[1] );
225  dst_points.push_back( mat * src_points[2] );
226  dst_points.push_back( mat * src_points[3] );
227  SetPoints( dst_points );
228  }
