104 // Parsing
105 /* Parse an FSpec from the macro input */
106 template< ::ULIS::e_tp iTp, ::ULIS::e_cm iCm, ::ULIS::e_ea iEa, ::ULIS::e_nm iNm >
107 constexpr FSpec parseSpec( const char* iSs, uint32_t iSh, const char* iCl )
108 {
109  constexpr bool dm = is_decimal( iTp );
110  uint8_t rc = ::__coal__::strlen( iCl );
111  uint8_t nc = iEa == ::ULIS::e_ea::khasAlpha ? rc - 1 : rc;
112  uint16_t pd = type_size< iTp >() * rc;
113  double tm = iNm == e_nm::knormalized ? 1.0 : type_max< iTp >();
114  double rm = dm ? tm : tm + 1;
115  uint8_t ma = 16;
116  uint32_t lh = COAL_CRC32_STR( iCl );
117  static_assert( iNm == e_nm::knormalized ? dm : true, "Integer types cannot be normalized" );
118  return { iSs, iSh, iTp, iCm, iEa, iCl, iNm, dm, rc, nc, pd, tm, rm, ma, lh };
119 }
