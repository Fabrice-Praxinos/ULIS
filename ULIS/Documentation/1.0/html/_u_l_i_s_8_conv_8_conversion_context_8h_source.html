146  // General Convert
147  template< uint32 _SHSrc, uint32 _SHDst >
148  static void Convert( const TPixelValue< _SHSrc >& iSrc, TPixelValue< _SHDst >& iDst )
149  {
150  using src_info = TBlockInfo< _SHSrc >;
151  using dst_info = TBlockInfo< _SHDst >;
152 
153  FConversionDiagnosis diag;
154  diag.bSameFormat = src_info::_nf._sh == dst_info::_nf._sh;
155  diag.bSameType = src_info::_nf._tp == dst_info::_nf._tp;
156  diag.bSameModel = src_info::_nf._cm == dst_info::_nf._cm;
157  diag.bSameLayout = src_info::_nf._lh == dst_info::_nf._lh;
158  diag.bSameProfile = iSrc.ColorProfile() == iDst.ColorProfile();
159 
160  if( diag.bSameFormat && diag.bSameProfile )
161  {
162  memcpy( iDst.Ptr(), iSrc.Ptr(), src_info::_nf._pd );
163  iDst.AssignColorProfile( iSrc.ColorProfile() );
164  return;
165  }
166 
167  using tSrcConnectionType = TPixelValue< TModelConnectionFormat< src_info::_nf._cm >() >;
168  using tDstConnectionType = TPixelValue< TModelConnectionFormat< dst_info::_nf._cm >() >;
169  tSrcConnectionType srcConnectionValue = TForwardConnector< _SHSrc, src_info::_nf._cm >::ConnectionModelFormat( iSrc );
170  tDstConnectionType dstConnectionValue = TForwardConnector< _SHDst, dst_info::_nf._cm >::ConnectionModelFormat( iDst );
171 
172  FColorProfile* src_profile = iSrc.ColorProfile();
173  FColorProfile* dst_profile = iDst.ColorProfile();
174  if( src_info::_nf._cm == dst_info::_nf._cm )
175  {
176  if( src_profile == nullptr ) src_profile = dst_profile;
177  if( dst_profile == nullptr ) dst_profile = src_profile;
178  }
179  else
180  {
181  if( src_profile == nullptr ) src_profile = FGlobalProfileRegistry::Get().GetDefaultProfileForModel( src_info::_nf._cm );
182  if( dst_profile == nullptr ) dst_profile = FGlobalProfileRegistry::Get().GetDefaultProfileForModel( dst_info::_nf._cm );
183  }
184 
185  cmsHPROFILE hInProfile = src_profile->ProfileHandle();
186  cmsHPROFILE hOutProfile = dst_profile->ProfileHandle();
187  cmsHTRANSFORM hTransform;
188  hTransform = cmsCreateTransform( hInProfile
189  , TCMSConnectionType< tSrcConnectionType::ColorModel() >()
190  , hOutProfile
191  , TCMSConnectionType< tDstConnectionType::ColorModel() >()
192  , INTENT_PERCEPTUAL, 0 );
193 
194 
195  cmsDoTransform( hTransform, srcConnectionValue.Ptr(), dstConnectionValue.Ptr(), 1 );
196 
197  TDropConnector< _SHDst, TModelConnectionFormat< dst_info::_nf._cm >(), dst_info::_nf._cm >::Apply( iDst, dstConnectionValue );
198  }
