109 // Standard Illuminant Chromaticity Pairs & Tristimulus Values conversion, xyY <=> XYZ
110 // This is equivalent to xyY -> XYZ
111 static inline glm::vec3 standardIlluminantTristimulusFromChromaticity( const glm::vec2& iVec )
112 {
113  float tempY = 1.f / iVec.y;
114  float X = tempY * iVec.x;
115  float Z = tempY * ( 1.f - iVec.x - iVec.y );
116  return glm::vec3( X, 1.f, Z );
117 }

120 // This is equivalent to XYZ -> xyY
121 static inline glm::vec2 standardIlluminantChromaticityFromTristimulus( const glm::vec3& iVec )
122 {
123  float tempSum = iVec.x + iVec.y + iVec.z;
124  float x = iVec.x / tempSum;
125  float y = iVec.y / tempSum;
126  return glm::vec2( x, y );
127 }
