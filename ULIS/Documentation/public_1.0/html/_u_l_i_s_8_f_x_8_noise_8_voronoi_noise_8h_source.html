64  static inline void Run( TBlock< _SH >* iBlock, uint32 iCount, int iSeed )
65  {
66  using tPixelProxy = typename TBlock< _SH >::tPixelProxy;
67  using info = TBlockInfo< _SH >;
68 
69  uint32 seed = iSeed < 0 ? time( NULL ) : iSeed;
70  std::minstd_rand generator( seed );
71 
72  const int fw = iBlock->Width() - 2;
73  const int fh = iBlock->Height() - 2;
74  std::vector< glm::vec2 > points( iCount );
75  for( int i = 0; i < iCount; ++i )
76  points[i] = glm::vec2( generator() % fw + 1, generator() % fh + 1 );
77 
78  float sqrcount = std::sqrt( (float)iCount );
79  float cellsizew = 1 * ( fw / sqrcount );
80  float cellsizeh = 1 * ( fh / sqrcount );
81  float normalisation_factor = std::sqrt( cellsizew * cellsizew + cellsizeh * cellsizeh );
82 
83  const int x1 = 0;
84  const int y1 = 0;
85  const int x2 = iBlock->Width();
86  const int y2 = iBlock->Height();
87  FThreadPool& global_pool = FGlobalThreadPool::Get();
88  for( int y = y1; y < y2; ++y )
89  global_pool.ScheduleJob( ProcessScanLine, iBlock, y, x1, x2, normalisation_factor, points );
90 
91  global_pool.WaitForCompletion();
92  }
